<script>
let chance = 0;
let total = 0;
let rolls = [];

// üî¢ GLOBAL CUSTOMER COUNT (same mobile continue)
let customer = localStorage.getItem("nb_customer_count")
  ? parseInt(localStorage.getItem("nb_customer_count"))
  : 0;

/*
WIN PLAN:
6  ‚Üí Soft Nugget Pav (15)
14 ‚Üí Momos (10)
25 ‚Üí Soft Popcorn (8)
33 ‚Üí Seekh Pav (20)
*/
const winPlan = {
  6 : { total:15, gift:"Soft Nugget Pav", code:"NB-SNPW15" },
  14: { total:10, gift:"Momos",          code:"NB-MSW10" },
  25: { total:8,  gift:"Soft Popcorn",   code:"NB-SPW8"  },
  33: { total:20, gift:"Seekh Pav",      code:"NB-SPVW20"}
};

// ‚ùå beech me ye totals kabhi nahi aane chahiye
const blockedTotals = [8,10,15,20];

function rollDice(){
  if(chance >= 3) return;

  document.getElementById("diceSound").play();

  const nextCustomer = customer + 1;
  let roll;

  if(!winPlan[nextCustomer]){
    roll = Math.floor(Math.random()*6)+1;
    if(chance === 2 && blockedTotals.includes(total + roll)){
      roll = 1;
    }
  }else{
    if(chance < 2){
      roll = Math.floor(Math.random()*6)+1;
    }else{
      roll = winPlan[nextCustomer].total - total;
      if(roll < 1 || roll > 6) roll = 1;
    }
  }

  rolls.push(roll);
  total += roll;
  chance++;

  const dice = document.getElementById("dice");
  dice.classList.add("roll");
  setTimeout(()=>dice.classList.remove("roll"),600);

  document.getElementById("rolls").innerText = rolls.join(", ");
  document.getElementById("total").innerText = total;

  if(chance === 3) checkResult();
}

function checkResult(){
  customer++;
  if(customer > 33) customer = 1;
  localStorage.setItem("nb_customer_count", customer);

  let msg = "üòÖ Better Luck Next Time";
  let html = "";

  if(winPlan[customer] && total === winPlan[customer].total){
    const w = winPlan[customer];
    msg = `üéâ Congratulations! You Won ${w.gift}`;
    html = `
      <div class="codeBox">
        üéüÔ∏è <b>Winning Code</b><br><br>
        <b style="font-size:22px">${w.code}</b><br><br>
        Order karte waqt message me ye code likhna
      </div>`;
    document.getElementById("winSound").play();
  }else{
    document.getElementById("loseSound").play();
  }

  document.getElementById("result").innerText = msg;
  document.getElementById("coupon").innerHTML = html;
  document.getElementById("nextBtn").style.display = "block";
}

function resetGame(){
  chance = 0;
  total = 0;
  rolls = [];
  document.getElementById("rolls").innerText = "‚Äî";
  document.getElementById("total").innerText = "0";
  document.getElementById("result").innerText = "";
  document.getElementById("coupon").innerHTML = "";
  document.getElementById("nextBtn").style.display = "none";
}
